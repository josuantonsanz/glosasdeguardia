import wiringpi
import time
from enum import Enum
from wiringpi import GPIO
from pynput.keyboard import Controller, Key
from abc import ABC, abstractmethod

keyboard = Controller()

class ControlMode(ABC):
    def __init__(self, controller):
        self.controller = controller

    @abstractmethod
    def click(self, button_index):
        """Maneja un click de botón en este modo"""
        pass

    def activate(self):
        """Llamado cuando se activa este modo"""
        pass

    def deactivate(self):
        """Llamado cuando se desactiva este modo"""
        pass

class RegisterMode(ControlMode):
    def __init__(self, controller):
        super().__init__(controller)
        self.led_states = [False] * 7
        self.register_keys = ['1', '2', '3', '4', '5', '6', '7']

    def click(self, button_index):
        # Cambiar estado del registro
        self.led_states[button_index] = not self.led_states[button_index]
        wiringpi.digitalWrite(self.controller.LED_PINS[button_index], 
                           1 if self.led_states[button_index] else 0)

        # Enviar tecla correspondiente
        keyboard.press(self.register_keys[button_index])
        time.sleep(0.05)
        keyboard.release(self.register_keys[button_index])

        print(f"Registro {button_index + 1} {'activado' if self.led_states[button_index] else 'desactivado'}")

class TransposeMode(ControlMode):
    def __init__(self, controller):
        super().__init__(controller)
        self.transpose_value = 0

    def click(self, button_index):
        if button_index == 1:
            self.transpose_value += 1
            print(f"Transposición: +{self.transpose_value} semitonos")
        elif button_index == 2:
            self.transpose_value -= 1
            print(f"Transposición: {self.transpose_value} semitonos")
        elif button_index == 6:
            self.transpose_value = 0
            print("Transposición reseteada")

    def activate(self):
        self.blink_all_leds()
        print(f"Modo Transposición activo. Valor actual: {self.transpose_value}")

    def blink_all_leds(self):
        for pin in self.controller.LED_PINS:
            wiringpi.digitalWrite(pin, 1)
        time.sleep(0.2)
        for pin in self.controller.LED_PINS:
            wiringpi.digitalWrite(pin, 0)

class VolumeMode(ControlMode):
    def __init__(self, controller):
        super().__init__(controller)
        self.volume_level = 100

    def click(self, button_index):
        if button_index == 1:
            self.volume_level = min(100, self.volume_level + 10)
            self.update_led_display()
            print(f"Volumen: {self.volume_level}%")
        elif button_index == 2:
            self.volume_level = max(0, self.volume_level - 10)
            self.update_led_display()
            print(f"Volumen: {self.volume_level}%")

    def update_led_display(self):
        num_leds = round((self.volume_level / 100.0) * len(self.controller.LED_PINS))
        for i in range(len(self.controller.LED_PINS)):
            wiringpi.digitalWrite(self.controller.LED_PINS[i], 1 if i < num_leds else 0)

    def activate(self):
        self.update_led_display()
        print(f"Modo Volumen activo. Nivel actual: {self.volume_level}%")

class OrganController:
    def __init__(self):
        # Configuración de pines
        self.LED_PINS = [3, 4, 5, 6, 7, 8, 9]
        self.BUTTON_PINS = [2, 10, 11, 12, 13, 14, 15]

        # Inicializar hardware
        wiringpi.wiringPiSetup()
        self.setup_pins()

        # Estado de los botones
        self.last_button_states = [True] * 7
        self.button_press_times = [0] * 7

        # Inicializar modos
        self.modes = {
            'register': RegisterMode(self),
            'transpose': TransposeMode(self),
            'volume': VolumeMode(self)
        }
        self.current_mode = self.modes['register']

        print("Controlador inicializado en modo Registros")

    def setup_pins(self):
        for pin in self.LED_PINS:
            wiringpi.pinMode(pin, GPIO.OUTPUT)
            wiringpi.digitalWrite(pin, 0)

        for pin in self.BUTTON_PINS:
            wiringpi.pinMode(pin, GPIO.INPUT)
            wiringpi.pullUpDnControl(pin, GPIO.PUD_UP)
